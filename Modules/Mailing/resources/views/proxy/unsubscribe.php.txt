    <script>
        async function confirmUnsubscribe() {
            const btn = document.getElementById('unsubscribe-btn');
            const initialState = document.getElementById('initial-state');
            const successState = document.getElementById('success-state');
            const errorState = document.getElementById('error-state');
            const errorMsg = document.getElementById('error-message');

            btn.disabled = true;
            btn.innerText = '...';

            console.log('Attempting unsubscription for:', '<?php echo $prospectId; ?>');

            try {
                const response = await fetch('<?php echo $backendUrl; ?>', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'Accept': 'application/json'
                    },
                    body: JSON.stringify({
                        prospect_id: '<?php echo $prospectId; ?>',
                        token: '<?php echo $token; ?>'
                    })
                });

                const data = await response.json();

                if (response.ok) {
                    initialState.classList.add('hidden');
                    successState.classList.remove('hidden');
                    document.querySelector('.success-icon').style.display = 'block';
                } else {
                    console.error('API Error Response:', data);
                    throw new Error(data.message || 'Error ' + response.status);
                }
            } catch (error) {
                console.error('Fetch Failed:', error);
                
                // Show more detailed error for troubleshooting
                // alert('DEBUG ERROR: ' + error.message); 

                initialState.classList.add('hidden');
                errorState.classList.remove('hidden');
                if (error.message === 'Invalid token.') {
                    errorMsg.innerText = 'De link is ongeldig of verlopen. / Le lien est invalide ou expiré.';
                } else if (error.message.includes('Failed to fetch')) {
                    errorMsg.innerText = 'Netwerkfout. Controleer of de backend bereikbaar is. / Erreur réseau.';
                }
            }
        }
    </script>
